image: maven:latest

stages:
  - build
  - test
  - package

variables:
  MAVEN_CLI_OPTS: " --batch-mode"
  MAVEN_OPTS: "-Dmaven.repo.local=./../.m2/repository"

cache:
  paths:
    - .m2/repository/

build:
  stage: build
  script:
    - cd prof-ap
    - mvn $MAVEN_CLI_OPTS compile
    - mvn $MAVEN_CLI_OPTS install
    - cd ..
    - cd prof-app
    - mvn $MAVEN_CLI_OPTS compile

test:
  stage: test
  script:
    - cd prof-app
    - mvn $MAVEN_CLI_OPTS test

package:
  stage: package
  script:
    - cd prof-app
    - mvn $MAVEN_CLI_OPTS package
  artifacts:
    paths:
    - prof-app/target/prof-app-0.0.1-SNAPSHOT-jar-with-dependencies.jar
    expire_in: 1 week
